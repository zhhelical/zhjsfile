function RenderCreator() {
	var pinnedPostCount = $('.waterfall-glist-container .item-pinned').length;
	var initialAllPostCount = $('.waterfall-glist-container .waterdrop').length;
	var insertRow = 0;//Math.max(Math.floor(pinnedPostCount/4),3);
	var sequenceId = 0;
	var ads;
	var colsizes;
	var startRow = 1;
	function randomItem(items) {
		return items[Math.floor(Math.random()*items.length)];
	}
	function randomColor() {
		var colors = ['#5ece76','#50b1e3','#f88788'];
		return colors[Math.floor(Math.random()*colors.length)];
	}
	function renderAd() {
		var ad = randomItem(ads);
		var template = $('#waterfall_adsense_tmpl').html();
		return $.renderAdsense({
			url: ad.curl,
			line1: ad.tit,
			line2: ad.desc,
			visible_url: ad.surl,
			color: randomColor()
		},template);
	}
	function initRender(_ads) {
		ads = []

		if (_ads) {
			for(var i = 0; i < _ads.length; i++) {
				if (_ads[i].surl.indexOf('58.com') === -1) {
					ads.push(_ads[i])
				}
			}
		}

		render();
	}
	function render(event, loadedAdSize) {
		if (loadedAdSize) {
			colsizes = $('.waterfall-glist').map(function(){
				return $(this).find('.waterdrop').length;
			});
		} else {
			colsizes = [-1,0,0,0,0];
			loadedAdSize = 20;
		}
		setTimeout(function(){
			var insertSize = Math.ceil(loadedAdSize/10) + 1;
			for(var i = 0; i < insertSize; i++ ) {
				var col = Math.floor(Math.random() * colsizes.length);
				if (colsizes[0]==-1) col = col+1;
				var insertRow = i * 3 + colsizes[col];
				var row = insertRow < startRow ? startRow : insertRow;
				var insertNode = $('.waterfall-glist').eq(col).find('.waterdrop').eq(row)
				insertNode.before( renderAd() );
			}
		}, loadedAdSize*100);
	}
	return {
		initRender: initRender,
		render: render
	};
}

	function b64EncodeUnicode(str) {
		return btoa(encodeURIComponent(str).replace(/%([0-9A-F]{2})/g, function(match, p1) {
			return String.fromCharCode('0x' + p1);
		}));
