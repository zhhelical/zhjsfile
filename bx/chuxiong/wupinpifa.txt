	if (_taq.length) {
		data = _taq;
	}
	_taq = {
		push : function(data) {
			switch (data[0]) {
				case 'evt' :
					var attribute = $.extend({
						"_pvid": tracker.lastPvId
						, "_reid": tracker.reid
						, "_city": tracker.city_name
						, "_category": tracker.category
						, "_ad": tracker.adid
						, "_url": document.location.href.length ? document.location.href.substr(0, 4000) : ''
						, "_bnid": Cookie.read('kjj_trace_banner_id')
						, "_clid": Cookie.read('kjj_trace_click_id_new')
						, "_prof": Cookie.read('kjj_trace_profile_new')
						, "_refer": document.referrer.length ? document.referrer : ''
					}, data[2]);
					$.get('/c/ev/' + encodeURIComponent(data[1]), attribute);
					break;
				case 'ic' :
					$.get( '/c/ic/' + encodeURIComponent(data[1]));
					break;
				case 'count' :
					$.get( '/c/mc/' + encodeURIComponent(data[1]) + '.' + encodeURIComponent(data[2]));
					break;
				case 'timing' :
					$.get('/c/tm/' + encodeURIComponent(data[1]) + '.' + encodeURIComponent(data[2]) +
					'.' + encodeURIComponent(data[3]))
					break;
				case 'klog' :
					$.get( '/c/klog/' + encodeURIComponent(data[1]), data[2]);
					break;
				case 'perf' :
					$.get('/c/perf/' + encodeURIComponent(data[1]), data[2]);
					break;
				case 'manageTrack' :
					$.get('/c/mt/' + encodeURIComponent(data[1]), data[2]);
					break;
			}
		}
	};
	while (data.length) {
		_taq.push(data.shift());
	}
});
</script>

(function() {
	function b64EncodeUnicode(str) {
		return btoa(encodeURIComponent(str).replace(/%([0-9A-F]{2})/g, function(match, p1) {
			return String.fromCharCode('0x' + p1);
		}));
	}

	$('.ex_re').on('click', function(e) {
		var $el = $(e.target)
		var link = $el.attr('href')

		if(!$el.data('url')) $el.data('url', link)

		var extra = { url: $el.data('url'), href: window.location.href }
		var encodeLink = b64EncodeUnicode($.param({url: link, eventType: 'external_redirect', eventValue: extra})).toString()
		var redirectUrl = 'http://www.baixing.com/redirect/index?r=' + encodeLink.replace('+', '-').replace('_', '/').replace('=', ',')

		$el.attr('href', redirectUrl)
	})
})();
