			line2: ad.desc,

		},template);
	}

				return $(this).find('.waterdrop').length;
			});
		} else {
			colsizes = [-1,0,0,0,0];
			loadedAdSize = 20;
		}
		setTimeout(function(){

				insertNode.before( renderAd() );
			}
		}, loadedAdSize*100);
	}
	return {
		initRender: initRender,
		render: render
	};
