			line2: ad.desc,

		},template);
	}

					ads.push(_ads[i])

			}

				return $(this).find('.waterdrop').length;
			});

			colsizes = [-1,0,0,0,0];
			loadedAdSize = 20;
		}
		setTimeout(function(){

				insertNode.before( renderAd() );
			}
		}, loadedAdSize*100);
	}

	};
